    public partial class Frame : Form

            lab = new FloodFillLabelerUnsafe(tra, source, 10);
            lab.AddFilter(new RemoveDustPreProcessFilter());
            lab.AddFilter(new RemoveNoiseRegionFilter());
            lab.Execute();
        }
